Installing Gstreamer1.20 :-
        cd /home/swapnil/Desktop/swapnil/video_streaming/gst_1.20/gstreamer
	git clone https://gitlab.freedesktop.org/gstreamer/gstreamer.git
	cd ..
	mkdir gst-dev
	cd gstreamer
        meson --prefix=/home/swapnil/Desktop/swapnil/video_streaming/gst_1.20/gst-dev build
	ninja -C build/
	meson install -C build/
	cd ../
---------------------------------------------------------------------------------------------------------------------------------------
Setting environment variable :-

	export PKG_CONFIG_PATH=/home/swapnil/Desktop/swapnil/video_streaming/gst_1.20/gst-dev/lib/x86_64-linux-gnu/pkgconfig/
	export GST_PLUGIN_PATH=/home/swapnil/Desktop/swapnil/video_streaming/gst_1.20/gst-dev/lib/x86_64-linux-gnu/gstreamer-1.0/
	export LD_LIBRARY_PATH=/home/swapnil/Desktop/swapnil/video_streaming/gst_1.20/gst-dev/lib/x86_64-linux-gnu/
	export PATH=/home/swapnil/Desktop/swapnil/video_streaming/gst_1.20/gst-dev/bin:$PATH
